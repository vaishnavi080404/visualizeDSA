// Stack Implementation in C++
#include <iostream>
using namespace std;

#define MAX 1000

class Stack {
    int top;
public:
    int arr[MAX];  // Stack array
    
    Stack() { top = -1; }  // Constructor
    
    // Push element onto stack
    bool push(int x) {
        if (top >= (MAX - 1)) {
            cout << "Stack Overflow";
            return false;
        }
        arr[++top] = x;
        return true;
    }
    
    // Pop element from stack
    int pop() {
        if (top < 0) {
            cout << "Stack Underflow";
            return 0;
        }
        return arr[top--];
    }
    
    // Peek top element
    int peek() {
        if (top < 0) {
            cout << "Stack is Empty";
            return 0;
        }
        return arr[top];
    }
    
    // Check if stack is empty
    bool isEmpty() {
        return (top < 0);
    }
};
# Stack Implementation in Python

class Stack:
    def __init__(self):
        # Initialize empty stack
        self.stack = []
    
    def push(self, item):
        # Add element to top
        self.stack.append(item)
    
    def pop(self):
        # Remove and return top element
        if self.isEmpty():
            return "Stack Underflow"
        return self.stack.pop()
    
    def peek(self):
        # Return top element without removing
        if self.isEmpty():
            return "Stack is Empty"
        return self.stack[-1]
    
    def isEmpty(self):
        # Check if stack is empty
        return len(self.stack) == 0
    
    def size(self):
        # Return number of elements
        return len(self.stack)
