📋 PSEUDO CODE
Step 1: Set low = 0, high = n-1
Step 2: While low ≤ high
Step 3: mid = (low + high) / 2
Step 4: If arr[mid] == target
Step 5: Return mid (Found!)
Step 6: If arr[mid] < target
Step 7: Set low = mid + 1
Step 8: Else set high = mid - 1
Step 9: Return -1 (Not found)

// Binary Search Implementation in C++
#include <iostream>
using namespace std;

// Function to perform binary search
int binarySearch(int array[], int x, int low, int high) {
    // Repeat until pointers meet
    while (low <= high) {
        // Find middle index
        int mid = low + (high - low) / 2;

        // Check if target is at mid
        if (array[mid] == x)
            return mid;

        // If target is greater, ignore left half
        else if (array[mid] < x)
            low = mid + 1;

        // If target is smaller, ignore right half
        else
            high = mid - 1;
    }

    // Element not found
    return -1;
}

int main() {
    int array[] = {3, 4, 5, 6, 7, 8, 9};
    int x = 4;
    int n = sizeof(array) / sizeof(array[0]);

    int result = binarySearch(array, x, 0, n - 1);
    
    (result == -1) ? cout << "Element not found" 
                    : cout << "Element found at index: " << result;
    
    return 0;
}
# Binary Search Implementation in Python

def binarySearch(array, x, low, high):
    # Repeat until the pointers low and high meet
    while low <= high:
        # Find middle index
        mid = low + (high - low) // 2

        # Check if target is at mid
        if array[mid] == x:
            return mid
        
        # If target is greater, ignore left half
        elif array[mid] < x:
            low = mid + 1
        
        # If target is smaller, ignore right half
        else:
            high = mid - 1
    
    # Element not found
    return -1

# Test the function
array = [3, 4, 5, 6, 7, 8, 9]
x = 4
result = binarySearch(array, x, 0, len(array) - 1)

if result != -1:
    print(f"Element found at index {result}")
else:
    print("Element not found")
