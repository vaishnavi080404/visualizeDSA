// Queue Implementation in C++ (Static Array approach)
#include <iostream>
using namespace std;

class Queue {
    int front, rear, capacity;
    int* queue;
public:
    // Constructor to initialize queue with specific size
    Queue(int c) { 
        front = rear = 0; 
        capacity = c; 
        queue = new int[c]; 
    }
    
    // Adds an element to the end (Rear)
    void enqueue(int data) {
        if (rear == capacity) { 
            cout << "Overflow: Cannot add to a full queue\n"; 
            return; 
        }
        queue[rear++] = data;
    }
    
    // Removes an element from the start (Front)
    void dequeue() {
        if (front == rear) { 
            cout << "Underflow: Cannot remove from empty queue\n"; 
            return; 
        }
        front++;
    }
};
# Queue Implementation in Python using List
class Queue:
    def __init__(self, limit):
        # Initialize internal list and size limit
        self.queue = []
        self.limit = limit
    
    def enqueue(self, item):
        # Check if queue has reached maximum capacity
        if len(self.queue) >= self.limit:
            return "Overflow"
        self.queue.append(item)
    
    def dequeue(self):
        # Check if queue is empty before popping
        if not self.isEmpty():
            return self.queue.pop(0)
        return "Underflow"

    def isEmpty(self):
        return len(self.queue) == 0
