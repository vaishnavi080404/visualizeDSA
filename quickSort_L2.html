// Quick Sort Implementation in C++
// Divide and conquer algorithm with in-place sorting

#include <iostream>
using namespace std;

/**
 * Swaps two elements in the array
 * @param a - pointer to first element
 * @param b - pointer to second element
 */
void swap(int* a, int* b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

/**
 * Partition function: places pivot in correct position
 * Elements smaller than pivot go to left, larger go to right
 * @param array - array to partition
 * @param low - starting index
 * @param high - ending index (pivot position)
 * @return final position of pivot
 */
int partition(int array[], int low, int high) {
    int pivot = array[high];  // Choose last element as pivot
    int i = (low - 1);        // Index of smaller element

    // Traverse through array and move smaller elements to left
    for (int j = low; j < high; j++) {
        if (array[j] <= pivot) {
            i++;
            swap(&array[i], &array[j]);
        }
    }
    
    // Place pivot in correct position
    swap(&array[i + 1], &array[high]);
    return (i + 1);
}

/**
 * Quick Sort: Recursively sorts array using divide and conquer
 * @param array - array to sort
 * @param low - starting index
 * @param high - ending index
 */
void quickSort(int array[], int low, int high) {
    if (low < high) {
        // Partition array and get pivot index
        int pi = partition(array, low, high);
        
        // Recursively sort left and right subarrays
        quickSort(array, low, pi - 1);   // Before pivot
        quickSort(array, pi + 1, high);  // After pivot
    }
}

// Example usage
int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);
    
    quickSort(arr, 0, n - 1);
    
    cout << "Sorted array: ";
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
    
    return 0;
}
# Quick Sort Implementation in Python
# Divide and conquer algorithm with in-place sorting

def partition(array, start, end):
    """
    Partition function using Hoare's scheme
    Places elements smaller than pivot on left, larger on right
    
    Args:
        array: List to partition
        start: Starting index
        end: Ending index
    
    Returns:
        Final position of pivot element
    """
    pivot = array[start]  # Choose first element as pivot
    low = start + 1
    high = end

    while True:
        # Move high pointer left until element <= pivot found
        while low <= high and array[high] >= pivot:
            high = high - 1

        # Move low pointer right until element >= pivot found
        while low <= high and array[low] <= pivot:
            low = low + 1

        # If pointers crossed, partition is complete
        if low <= high:
            array[low], array[high] = array[high], array[low]
        else:
            break

    # Place pivot in final position
    array[start], array[high] = array[high], array[start]
    return high


def quick_sort(array, start, end):
    """
    Quick Sort: Recursively sorts array using divide and conquer
    
    Args:
        array: List to sort
        start: Starting index
        end: Ending index
    """
    if start >= end:
        return  # Base case: single element or empty

    # Partition array and get pivot position
    p = partition(array, start, end)
    
    # Recursively sort left and right subarrays
    quick_sort(array, start, p - 1)  # Before pivot
    quick_sort(array, p + 1, end)    # After pivot


# Example usage
if __name__ == "__main__":
    arr = [64, 34, 25, 12, 22, 11, 90]
    print(f"Original array: {arr}")
    
    quick_sort(arr, 0, len(arr) - 1)
    
    print(f"Sorted array: {arr}")
